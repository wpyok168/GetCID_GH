// ========================= 页面交互 ========================= const $ = id => document.getElementById(id); const iidInput = $("iidInput"); const getBtn = $("getBtn"); const resultBox = $("resultBox"); const copyBtn = $("copyBtn"); const status = $("status"); // 👇 这里填你 Cloudflare Worker 分配的地址 const PROXY_URL = "https://你的-worker-name-xxxx.your-subdomain.workers.dev/"; getBtn.addEventListener("click", async () => { try { getBtn.disabled = true; getBtn.textContent = "处理中..."; status.textContent = "🔍 校验 IID..."; const raw = iidInput.value.trim(); // 简单清洗 IID const iid = (raw || "").replace(/[^0-9]/g, ""); if (!iid) { toast("请输入IID"); throw new Error("IID为空"); } // 简单长度校验 if(iid.length !== 54 && iid.length !== 63) { toast("IID长度必须是 54 或 63 位"); throw new Error("长度错误"); } status.textContent = "📡 请求代理服务..."; // 1. 只向后端发送 IID，所有复杂计算都在 Cloudflare 完成 const response = await fetch(PROXY_URL, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ IID: iid }) }); const result = await response.json(); if (!response.ok) throw new Error(result.error || "请求失败"); // 2. 展示结果 resultBox.value = JSON.stringify(result, null, 2); copyBtn.disabled = false; toast("✅ 获取成功！"); status.textContent = "✅ 完成"; } catch (err) { toast("❌ 错误：" + err.message); status.textContent = "❌ 失败"; console.error(err); } finally { getBtn.disabled = false; getBtn.textContent = "获取确认 ID"; } }); // 复制功能 copyBtn.addEventListener("click", () => { resultBox.select(); navigator.clipboard.writeText(resultBox.value).then(() => { toast("已复制"); }); }); // Toast 提示 function toast(msg) { const t = document.getElementById('toast'); t.textContent = msg; t.classList.add('show'); setTimeout(() => t.classList.remove('show'), 2500); }
